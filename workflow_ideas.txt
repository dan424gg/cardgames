name: Flutter Release Workflow

on:
  push:
    branches:
      - main
  workflow_dispatch: # Allows manual trigger from GitHub Actions tab

jobs:
  # Job to run tests and basic checks
  test_and_analyze:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Setup Flutter SDK
        uses: subosito/flutter-action@v2
        with:
          channel: 'stable'
      - name: Install dependencies
        run: flutter pub get
      - name: Run tests
        run: flutter test
      - name: Analyze project source
        run: dart analyze

  # Job for building and deploying Android
  build_android:
    needs: test_and_analyze
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Setup Flutter SDK
        uses: subosito/flutter-action@v2
        with:
          channel: 'stable'
      # Steps to set up Java and decode keystore from GitHub secrets would go here
      - name: Build Android APK
        run: flutter build apk --release
      # Example deploy step:
      # - name: Deploy to Firebase App Distribution
      #   uses: FirebaseExtended/github-actions@v8
      #   with:
      #     ...

  # Job for building and deploying iOS
  build_ios:
    needs: test_and_analyze
    # iOS builds must run on a macOS runner
    runs-on: macos-latest 
    steps:
      - uses: actions/checkout@v4
      - name: Setup Flutter SDK
        uses: subosito/flutter-action@v2
        with:
          channel: 'stable'
      # Steps for Xcode setup, certificate handling, and code signing would go here
      - name: Build iOS app (no codesign for CI testing, actual deploy needs signing)
        run: flutter build ios --release --no-codesign
      # Example deploy step (often uses Fastlane or similar tools):
      # - name: Deploy to TestFlight/App Store
      #   uses: some-deploy-action@v1
      #   with:
      #     ...

  # Job for building and deploying Web
  build_web:
    needs: test_and_analyze
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Setup Flutter SDK
        uses: subosito/flutter-action@v2
        with:
          channel: 'stable'
      - name: Enable web support
        run: flutter config --enable-web
      - name: Build Web release
        run: flutter build web --release
